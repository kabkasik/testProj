<script type="text/javascript">
    $(function(){
        $( ".quotes .overview" ).rcarousel({
            width:960,
            height:145,
            visible:1,
            step:1,
            navigation: {
                prev: ".quotes .prev",
                next: ".quotes .next"
            }
        });
        $( "#hero .overview" ).rcarousel({
            width:960,
            height:485,
            visible:1,
            step:1,
            speed:0,
            start: generatePages,
            pageLoaded: pageLoaded,
            navigation: {
                prev: ".hero .prev",
                next: ".hero .next"
            }
        });
        if (navigator.appVersion.indexOf("Mac")!=-1){
            $('body').attr('id','mac');
        }else{
            $('body').attr('id','win');
        }
        $('.features li:odd').addClass('alt');
        $('.cta a').hover(function(){
          if (window.devicePixelRatio == 2) {
            $('body').css('background-image', 'url(./images/tree_animation_on@2x.gif)');
          }
          else {
            $('body').css('background-image', 'url(./images/tree_animation_on.gif)');
          }
          
        },
        function(){
          if (window.devicePixelRatio == 2) {
            $('body').css('background-image', 'url(./images/tree_animation_off@2x.gif)');
          }
          else {
            $('body').css('background-image', 'url(./images/tree_animation_off.gif)');
          }
        });
        preload([
            'images/tree_animation_off.gif',
            'images/tree_animation_on.gif',
            'images/tree_animation_off@2x.gif',
            'images/tree_animation_on@2x.gif'
        ]);
    });
    function preload(arrayOfImages) {
        $(arrayOfImages).each(function(){
            $('<img/>')[0].src = this;
        });
    }
    function generatePages() {
        var _total, i, _link, pagerWidth;
        _total = $("#hero .overview").rcarousel("getTotalPages");
        pagerWidth = _total*21;
        for (i = 0; i < _total; i++) {
            _link = $("<a href='#'></a>");
            $(_link)
                    .bind("click", {page: i},
                    function(event) {
                        $("#hero .overview").rcarousel("goToPage", event.data.page);
                        event.preventDefault();
                    }
            )
                    .addClass("bullet off")
                    .appendTo(".pager");
        }
        // mark first page as active
        $("a:eq(0)", ".pager")
                .removeClass("off")
                .addClass("on");
        $('.pager').css({'width': pagerWidth,'display':'block'});
    }
    function pageLoaded(event, data) {
        $("a.on", ".pager")
                .removeClass("on");
        $("a", ".pager")
                .eq(data.page)
                .addClass("on");
    }
</script>
